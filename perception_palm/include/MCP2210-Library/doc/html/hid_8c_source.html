00123 /* uses_numbered_reports() returns 1 if report_descriptor describes a device
00124    which contains numbered reports. */ 
00125 static int uses_numbered_reports(__u8 *report_descriptor, __u32 size) {
00126         int i = 0;
00127         int size_code;
00128         int data_len, key_size;
00129         
00130         while (i < size) {
00131                 int key = report_descriptor[i];
00132 
00133                 /* Check for the Report ID key */
00134                 if (key == 0x85/*Report ID*/) {
00135                         /* This device has a Report ID, which means it uses
00136                            numbered reports. */
00137                         return 1;
00138                 }
00139                 
00140                 //printf("key: %02hhx\n", key);
00141                 
00142                 if ((key & 0xf0) == 0xf0) {
00143                         /* This is a Long Item. The next byte contains the
00144                            length of the data section (value) for this key.
00145                            See the HID specification, version 1.11, section
00146                            6.2.2.3, titled "Long Items." */
00147                         if (i+1 < size)
00148                                 data_len = report_descriptor[i+1];
00149                         else
00150                                 data_len = 0; /* malformed report */
00151                         key_size = 3;
00152                 }
00153                 else {
00154                         /* This is a Short Item. The bottom two bits of the
00155                            key contain the size code for the data section
00156                            (value) for this key.  Refer to the HID
00157                            specification, version 1.11, section 6.2.2.2,
00158                            titled "Short Items." */
00159                         size_code = key & 0x3;
00160                         switch (size_code) {
00161                         case 0:
00162                         case 1:
00163                         case 2:
00164                                 data_len = size_code;
00165                                 break;
00166                         case 3:
00167                                 data_len = 4;
00168                                 break;
00169                         default:
00170                                 /* Can't ever happen since size_code is & 0x3 */
00171                                 data_len = 0;
00172                                 break;
00173                         };
00174                         key_size = 1;
00175                 }
00176                 
00177                 /* Skip over this key and it's associated data */
00178                 i += data_len + key_size;
00179         }
00180         
00181         /* Didn't find a Report ID key. Device doesn't use numbered reports. */
00182         return 0;
00183 }
